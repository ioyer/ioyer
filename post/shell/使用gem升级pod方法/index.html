升级cocopods

sudo gem update --system
gem source -l
pod setup
pod repo update --verbose
sudo gem install cocoapods --pre
sudo gem cleanup

gem source

$ gem sources --add https://gems.ruby-china.com/ --remove https://rubygems.org/
$ gem sources -l
https://gems.ruby-china.com
# 确保只有 gems.ruby-china.com
